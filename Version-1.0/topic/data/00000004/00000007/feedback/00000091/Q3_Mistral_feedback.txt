```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("請輸入數組的長度:");
        int n = sc.nextInt();
        int[] arr = new int[n];

        System.out.println("請輸入數組元素:");
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        replaceElements(arr);

        System.out.println("替換後的數組:");
        for (int i = 0; i < n; i++) {
            System.out.print(arr[i] + " ");
        }
    }

    public static void replaceElements(int[] arr) {
        int max = -1;
        for (int i = arr.length - 1; i >= 0; i--) {
            int temp = arr[i];
            arr[i] = max;
            if (temp > max) {
                max = temp;
            }
        }
    }
}
```
