### 3. 改進的程式碼

```java
// 1018上課練習_3
import java.util.Scanner;

public class a1018_3 {
    public static void main(String arg[]) {
        Scanner scn = new Scanner(System.in);
        System.out.print("請輸入數字個數: ");
        int n = scn.nextInt();
        int[] arr = new int[n];

        System.out.print("請輸入數字: ");
        for (int i = 0; i < n; i++) {
            arr[i] = scn.nextInt();
        }

        // 找到每個元素右邊的最大值
        for (int i = 0; i < n; i++) {
            int maxRight = -1;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] > maxRight) {
                    maxRight = arr[j];
                }
            }
            arr[i] = maxRight;
        }

        // 最後一個元素設為 -1
        arr[n - 1] = -1;

        // 輸出結果
        System.out.print("輸出: ");
        for (int i = 0; i < n; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }
}
```
