public class MyMatrix {
    private int[][] matrix;

    public MyMatrix(int rows, int cols) {
        this.matrix = new int[rows][cols];
    }

    public MyMatrix(int[][] values) {
        this.matrix = values;
    }

    public static MyMatrix add(MyMatrix m1, MyMatrix m2) {
        if (m1.matrix.length != m2.matrix.length || m1.matrix[0].length != m2.matrix[0].length) {
            throw new IllegalArgumentException("矩陣必須具有相同的尺寸才能相加。");
        }

        int rows = m1.matrix.length;
        int cols = m1.matrix[0].length;
        int[][] result = new int[rows][cols];

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                result[i][j] = m1.matrix[i][j] + m2.matrix[i][j];
            }
        }

        return new MyMatrix(result);
    }

    //其他函式的描述與程式碼

    public void printMatrix() {
        for (int[] row : matrix) {
            for (int element : row) {
                System.out.print(element + " ");
            }
            System.out.println();
        }
    }

    public static void main(String[] args) {
        //程式碼
    }
}
